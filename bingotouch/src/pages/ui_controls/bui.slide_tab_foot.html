    html:

    <div id="tabNews" class="bui-tab"> <div class="bui-tab-main"> <ul> <li> 滑动的内容 </li> <li> 滑动的内容 </li> </ul> </div> </div> <!-- 菜单在外层 要用id --> <ul id="tabNav" class="bui-nav"> <li class="bui-btn active"> TAB1 </li> <li class="bui-btn"> TAB2 </li> </ul>

    var tab = bui.slide({ id:"#tabNews", menu:"#tabNav", children:".bui-tab-main > ul", scroll: true })
